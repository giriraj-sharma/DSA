class Solution {
public:
    // TC: O(n)
    // SC: O(1)
    int maxProfit(vector<int>& prices) {

        int minPrice = prices[0];
        int maxProfit = 0;

        for (int i = 1; i < prices.size(); i++) {
            minPrice = min(minPrice, prices[i]);          // best buying price
            maxProfit = max(maxProfit, prices[i] - minPrice); // best profit
        }

        return maxProfit;
    }
};
