class Solution {
public:
    // TC: O(n)
    // SC: O(n)
    bool isValid(string s) {

        stack<char> st;

        for (char ch : s) {

            // Push opening brackets
            if (ch == '(' || ch == '{' || ch == '[') {
                st.push(ch);
            }
            // Handle closing brackets
            else {
                if (st.empty())
                    return false;

                if ((ch == ')' && st.top() != '(') ||
                    (ch == '}' && st.top() != '{') ||
                    (ch == ']' && st.top() != '['))
                    return false;

                st.pop();
            }
        }

        // Stack should be empty at the end
        return st.empty();
    }
};
